151 static void cmd_sync(int readfd, int writefd) {
152  char syncbuf[1];
153 
154  // first sync the device and the server.
155  // I.e. we send an FS_SYNC byte, and wait for the sync byte return,
156  // ignoreing all other bytes
157  //
158  // write the sync byte
159  syncbuf[0] = FS_SYNC;
160  int e = 0;
161  do {
162 #if defined DEBUG_WRITE || defined DEBUG_CMD
163  printf("sync write %02x\n", syncbuf[0]);
164 #endif
165  e = write(writefd, syncbuf, 1);
166  } while (e == 0 || (e < 0 && errno == EAGAIN));
167  if (e < 0) {
168  log_errno("Error on sync write", errno);
169  }
170  // wait for a sync byte
171  int n = 0;
172  do {
173  n = read(readfd, syncbuf, 1);
174 #ifdef DEBUG_READ
175  printf("sync read %d bytes: ",n);
176  for(int i=0;i<n;i++) printf("%02x ",255&syncbuf[i]); printf("\n");
177 #endif
178  if (n < 0) {
179  log_errno("Error on sync read", errno);
180  }
181  } while (n != 1 || (0xff & syncbuf[0]) != FS_SYNC);
182 }

191 void cmd_loop(int readfd, int writefd) {
192 
193  char buf[8192];
194  int wrp,rdp,plen, cmd;
195  int n;
196 
197  // sync device and server
198  cmd_sync(readfd, writefd);
199 
200  /* write and read pointers in the input buffer "buf" */
201  wrp = rdp = 0;
202 
203  while((n=read(readfd, buf+wrp, 8192-wrp))!=0) {
204 #ifdef DEBUG_READ
205  printf("read %d bytes: ",n);
206  for(int i=0;i<n;i++) printf("%02x ",255&buf[wrp+i]); printf("\n");
207 #endif
208 
209  if(n<0) {
210  fprintf(stderr,"fstcp: read error %d (%s)\n",errno,strerror(errno));
211  break;
212  }
213  wrp+=n;
214  if(rdp && (wrp==8192 || rdp==wrp)) {
215  if(rdp!=wrp) {
216  memmove(buf, buf+rdp, wrp-rdp);
217  }
218  wrp -= rdp;
219  rdp = 0;
220  }
221 //printf("wrp=%d, rdp=%d, FSP_LEN=%d\n",wrp,rdp,FSP_LEN);
222  // as long as we have more than FSP_LEN bytes in the buffer
223  // i.e. 2 or more, we loop and process packets
224  // FSP_LEN is the position of the packet length
225  while(wrp-rdp > FSP_LEN) {
226  // first byte in packet is command, second is length of packet
227  plen = 255 & buf[rdp+FSP_LEN]; // AND with 255 to fix sign
228  cmd = 255 & buf[rdp+FSP_CMD]; // AND with 255 to fix sign
229 //printf("wrp-rdp=%d, plen=%d\n",wrp-rdp,plen);
230  // full packet received already?
231  if (cmd == FS_SYNC || plen < FSP_DATA) {
232  // a packet is at least 3 bytes (when with zero data length)
233  // or the byte is the FS_SYNC command
234  // so ignore byte and shift one in buffer position
235  rdp++;
236  } else
237  if(wrp-rdp >= plen) {
238  // yes, then execute
239  do_cmd(buf+rdp, writefd);
240  rdp +=plen;
241  } else {
242  break;
243  }
244  }
245  }

256 static void do_cmd(char *buf, int fd) {
257  int tfd, cmd;
258  unsigned int len;
259  char retbuf[200];
260  int rv;
261 
262  cmd = buf[FSP_CMD]; // 0
263  len = 255 & buf[FSP_LEN]; // 1
264 
265  if (cmd == FS_TERM) {
266 #ifdef DEBUG_CMD_TERM
267  {
268  int n = buf[FSP_LEN];
269  log_debug("term: %d bytes @%p: ",n, buf);
270  for(int i=0;i<n;i++) printf("%02x ",255&buf[i]); printf("\n");
271  }
272 #endif
273 
274  if (len > 200) {
275  log_error("TERM length exceeds 200! Is %d\n", len);
276  len = 200;
277  }
278  memcpy(retbuf, buf+3, len-3);
279  retbuf[len-3] = 0;
280  log_term(retbuf);
281  return;
282  }
283 
284 #ifdef DEBUG_CMD
285  {
286  int n = buf[FSP_LEN];
287  log_debug("cmd: %d bytes @%p: ",n, buf);
288  for(int i=0;i<n;i++) printf("%02x ",255&buf[i]); printf("\n");
289  }
290 #endif
291 
292  // not on FS_TERM
293  tfd = buf[FSP_FD];
294  if (tfd < 0) {
295  printf("Illegal file descriptor: %d\n", tfd);
296  return;
297  }
298 
299  buf[(unsigned int)buf[FSP_LEN]] = 0; // 0-terminator
300 
301  provider_t *prov = NULL; //&fs_provider;
302  endpoint_t *ep = NULL;
303 
304 #if 0
305  printf("got cmd=%d, fd=%d, name=%s",cmd,tfd,
306  (cmd<FS_ASSIGN)?((char*)buf+FSP_DATA):"null");
307  for (int i = 3; i < buf[FSP_LEN]; i++) printf(" %02x", (unsigned int)buf[i]);
308  printf("\n");
309 #endif
310  retbuf[FSP_CMD] = FS_REPLY;
311  retbuf[FSP_LEN] = 4;
312  retbuf[FSP_FD] = tfd;
313  retbuf[FSP_DATA] = ERROR_FAULT;
314 
315  int eof = 0;
316  int outdeleted = 0;
317 
318  switch(cmd) {
319  // file-oriented commands
320  case FS_OPEN_WR:
321  ep = provider_lookup(buf[FSP_DATA]);
322  if (ep != NULL) {
323  prov = (provider_t*) ep->ptype;
324  rv = prov->open_wr(ep, tfd, buf + FSP_DATA + 1);
325  retbuf[FSP_DATA] = rv;
326  if (rv == 0) {
327  set_chan(tfd, ep);
328  } else {
329  log_rv(rv);
330  }
331  }
332  break;
333  case FS_OPEN_DR:
334  ep = provider_lookup(buf[FSP_DATA]);
335  if (ep != NULL) {
336  prov = (provider_t*) ep->ptype;
337  // not all providers support directory operation
338  if (prov->opendir != NULL) {
339  rv = prov->opendir(ep, tfd, buf + FSP_DATA + 1);
340  retbuf[FSP_DATA] = rv;
341  if (rv == 0) {
342  set_chan(tfd, ep);
343  } else {
344  log_rv(rv);
345  }
346  }
347  }
348  break;
349  case FS_OPEN_RD:
350  ep = provider_lookup(buf[FSP_DATA]);
351  if (ep != NULL) {
352  prov = (provider_t*) ep->ptype;
353  rv = prov->open_rd(ep, tfd, buf + FSP_DATA + 1);
354  retbuf[FSP_DATA] = rv;
355  if (rv == 0) {
356  set_chan(tfd, ep);
357  } else {
358  log_rv(rv);
359  }
360  }
361  break;
362  case FS_OPEN_AP:
363  ep = provider_lookup(buf[FSP_DATA]);
364  if (ep != NULL) {
365  prov = (provider_t*) ep->ptype;
366  rv = prov->open_ap(ep, tfd, buf + FSP_DATA + 1);
367  retbuf[FSP_DATA] = rv;
368  if (rv == 0) {
369  set_chan(tfd, ep);
370  } else {
371  log_rv(rv);
372  }
373  }
374  break;
375  case FS_READ:
376  ep = chan_to_endpoint(tfd);
377  if (ep != NULL) {
378  eof = 0; // default just in case
379  prov = (provider_t*) ep->ptype;
380  rv = prov->readfile(ep, tfd, retbuf + FSP_DATA, MAX_BUFFER_SIZE-FSP_DATA, &eof);
381  // TODO: handle error (rv<0)
382  if (rv < 0) {
383  log_rv(-rv);
384  }
385  retbuf[FSP_LEN] = FSP_DATA + rv;
386  if (eof) {
387  retbuf[FSP_CMD] = FS_EOF;
388  }
389  }
390  break;
391  case FS_WRITE:
392  case FS_EOF:
393  ep = chan_to_endpoint(tfd);
394  //printf("WRITE: chan=%d, ep=%p\n", tfd, ep);
395  if (ep != NULL) {
396  prov = (provider_t*) ep->ptype;
397  rv = prov->writefile(ep, tfd, buf+FSP_DATA, len-FSP_DATA, cmd == FS_EOF);
398  if (rv != 0) {
399  log_rv(rv);
400  }
401  retbuf[FSP_DATA] = rv;
402  }
403  break;
404  case FS_CLOSE:
405  ep = chan_to_endpoint(tfd);
406  if (ep != NULL) {
407  prov = (provider_t*) ep->ptype;
408  prov->close(ep, tfd);
409  free_chan(tfd);
410  retbuf[FSP_DATA] = ERROR_OK;
411  }
412  break;
413 
414  // command operations
415  case FS_DELETE:
416  ep = provider_lookup(buf[FSP_DATA]);
417  if (ep != NULL) {
418  prov = (provider_t*) ep->ptype;
419  if (prov->scratch != NULL) {
420  rv = prov->scratch(ep, buf+FSP_DATA+1, &outdeleted);
421  if (rv == ERROR_SCRATCHED) {
422  retbuf[FSP_DATA + 1] = outdeleted > 99 ? 99 :outdeleted;
423  retbuf[FSP_LEN] = FSP_DATA + 2;
424  } else
425  if (rv != 0) {
426  log_rv(rv);
427  }
428  retbuf[FSP_DATA] = rv;
429  }
430  }
431  break;
432  case FS_RENAME:
433  ep = provider_lookup(buf[FSP_DATA]);
434  if (ep != NULL) {
435  prov = (provider_t*) ep->ptype;
436  if (prov->rename != NULL) {
437  rv = prov->rename(ep, buf+FSP_DATA+1);
438  if (rv != 0) {
439  log_rv(rv);
440  }
441  retbuf[FSP_DATA] = rv;
442  }
443  }
444  break;
445  case FS_CHDIR:
446  ep = provider_lookup(buf[FSP_DATA]);
447  if (ep != NULL) {
448  prov = (provider_t*) ep->ptype;
449  if (prov->cd != NULL) {
450  rv = prov->cd(ep, buf+FSP_DATA+1);
451  if (rv != 0) {
452  log_rv(rv);
453  }
454  retbuf[FSP_DATA] = rv;
455  }
456  }
457  break;
458  case FS_MKDIR:
459  ep = provider_lookup(buf[FSP_DATA]);
460  if (ep != NULL) {
461  prov = (provider_t*) ep->ptype;
462  if (prov->mkdir != NULL) {
463  rv = prov->mkdir(ep, buf+FSP_DATA+1);
464  if (rv != 0) {
465  log_rv(rv);
466  }
467  retbuf[FSP_DATA] = rv;
468  }
469  }
470  break;
471  case FS_RMDIR:
472  ep = provider_lookup(buf[FSP_DATA]);
473  if (ep != NULL) {
474  prov = (provider_t*) ep->ptype;
475  if (prov->rmdir != NULL) {
476  rv = prov->rmdir(ep, buf+FSP_DATA+1);
477  if (rv != 0) {
478  log_rv(rv);
479  }
480  retbuf[FSP_DATA] = rv;
481  }
482  }
483  break;
484  case FS_ASSIGN:
485  // assign an endpoint number (i.e. a drive number for the PET)
486  // to a filesystem provider and path
487  // The drive number in buf[FSP_DATA] is the one to assign,
488  // while the rest of the name determines which provider to use
489  //
490  // A provider can be determines relative to an existing one. In
491  // this case the provider name is the endpoint (drive) number,
492  // and the path is interpreted as relative to an existing endpoint.
493  // If the provider name is a real name, the path is absolute.
494  //
495  rv = provider_assign(buf[FSP_DATA], buf+FSP_DATA+1);
496  if (rv != 0) {
497  log_rv(rv);
498  }
499  retbuf[FSP_DATA] = rv;
500  break;
501  }
502 
503 
504  int e = write(fd, retbuf, 0xff & retbuf[FSP_LEN]);
505  if (e < 0) {
506  printf("Error on write: %d\n", errno);
507  }
508 #if defined DEBUG_WRITE || defined DEBUG_CMD
509  printf("write %02x %02x %02x:", 255&retbuf[0], 255&retbuf[1],
510  255&retbuf[2] );
511  for (int i = 3; i<retbuf[FSP_LEN];i++) printf(" %02x", 255&retbuf[i]);
512  printf("\n");
513 #endif
514 }
