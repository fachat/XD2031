128 static endpoint_t *fsp_new(endpoint_t *parent, const char *path) {
129 
130  fs_endpoint_t *parentep = (fs_endpoint_t*) parent;
131 
132  // alloc and init a new endpoint struct
133  fs_endpoint_t *fsep = malloc(sizeof(fs_endpoint_t));
134  fsep->basepath = NULL;
135  fsep->curpath = NULL;
136  for(int i=0;i<MAXFILES;i++) {
137  init_fp(&(fsep->files[i]));
138  }
139 
140  fsep->ptype = (struct provider_t *) &fs_provider;
141 
142  char *dirpath = malloc_path((parentep == NULL) ? NULL : parentep->curpath,
143  path);
144 
145  // malloc's a buffer and stores the canonical real path in it
146  fsep->basepath = realpath(dirpath, NULL);
147 
148  free(dirpath);
149 
150  if (fsep->basepath == NULL) {
151  // some problem with dirpath - maybe does not exist...
152  log_errno("Could not resolve path for assign");
153  fsp_free((endpoint_t*)fsep);
154  return NULL;
155  }
156 
157  if (parentep != NULL) {
158  // if we have a parent, make sure we do not
159  // escape the parent container, i.e. basepath
160  if (strstr(fsep->basepath, parentep->basepath) != fsep->basepath) {
161  // the parent base path is not at the start of the new base path
162  // so we throw an error
163  log_error("ASSIGN broke out of container (%s), was trying %s\n",
164  parentep->basepath, fsep->basepath);
165  fsp_free((endpoint_t*)fsep);
166 
167  return NULL;
168  }
169  }
170 
171  // copy into current path
172  fsep->curpath = malloc(strlen(fsep->basepath) + 1);
173  strcpy(fsep->curpath, fsep->basepath);
174 
175  log_info("FS provider set to real path '%s'\n", fsep->basepath);
176 
177  return (endpoint_t*) fsep;
178 }

217  fs_endpoint_t *cep = (fs_endpoint_t*) ep;
218 
219  for (int i = 0; i < MAXFILES; i++) {
220  if (cep->files[i].chan == chan) {

227 
228  //log_debug("reserving file %p for chan %d\n", fp, chan);
229 
230  return &(cep->files[i]);
231  }
232  }
233  log_warn("Did not find free curl session for channel=%d\n", chan);
234  return NULL;

326 // read directory
327 //
328 // Note: there is a race condition, as we do not save the current directory path
329 // on directory open, so if it is changed "in the middle" of the operation,
330 // we run into trouble here. Hope noone will do that...
331 //
332 // returns the number of bytes read (>= 0), or a negative error number
333 //
334 static int read_dir(endpoint_t *ep, int tfd, char *retbuf, int len, int *eof) {
335 
336  fs_endpoint_t *fsep = (fs_endpoint_t*) ep;
337 
338  File *file = find_file(ep, tfd);
339 
340  //log_debug("read_dir: file=%p, is_first=%d\n", file, (file == NULL) ? -1 : file->is_first);
341 
342  if (file != NULL) {
343  int rv = 0;
344  if (file->is_first) {
345  file->is_first = 0;
346  int l = dir_fill_header(retbuf, 0, file->dirpattern);
347  rv = l;
348  file->de = dir_next(file->dp, file->dirpattern);
349  return rv;
350  }
351  if(!file->de) {
352  close_fds(ep, tfd);
353  *eof = 1;
354  int l = dir_fill_disk(retbuf);
355  rv = l;
356  return rv;
357  }
358  int l = dir_fill_entry(retbuf, fsep->curpath, file->de, len);
359  rv = l;
360  // prepare for next read (so we know if we're done)
361  file->de = dir_next(file->dp, file->dirpattern);
362  return rv;
363  }
364  return -ERROR_FAULT;
365 }

367 // read file data
368 //
369 // returns positive number of bytes read, or negative error number
370 //
371 static int read_file(endpoint_t *ep, int tfd, char *retbuf, int len, int *eof) {
372  File *file = find_file(ep, tfd);
373 
374  if (file != NULL) {
375  int rv = 0;
376 
377  FILE *fp = file->fp;
378 
379  int n = fread(retbuf, 1, len, fp);
380  rv = n;
381  if(n<len) {
382  // short read, so either error or eof
383  *eof = 1;
384  close_fds(ep, tfd);
385  } else {
386  // as feof() does not let us know if the file is EOF without
387  // having attempted to read it first, we need this kludge
388  int eofc = fgetc(fp);
389  if (eofc < 0) {
390  // EOF
391  *eof = 1;
392  //printf("Setting EOF!\n");
393  close_fds(ep, tfd);
394  } else {
395  // restore fp, so we can read it properly on the next request
396  ungetc(eofc, fp);
397  // do not send EOF
398  }
399  }
400  return rv;
401  }
402  return -ERROR_FAULT;
403 }

449 static int fs_delete(endpoint_t *ep, char *buf, int *outdeleted) {
450 
451  int matches = 0;
452  char *p = buf;
453 
454  fs_endpoint_t *fsep = (fs_endpoint_t*) ep;
455 
456  do {
457  // comma is file pattern separator
458  char *pnext = index(p, ',');
459  if (pnext != NULL) {
460  *pnext = 0; // write file name terminator (replacing the ',')
461  }
462 
463  int rv = dir_call_matches(fsep->curpath, p, _delete_callback);
464  if (rv < 0) {
465  // error happened
466  return -rv;
467  }
468  matches += rv;
469 
470  p = (pnext == NULL) ? NULL : pnext+1;
471  }
472  while (p != NULL);
473 
474  *outdeleted = matches;
475 
476  return ERROR_SCRATCHED; // FILES SCRATCHED message
477 }

479 static int fs_rename(endpoint_t *ep, char *buf) {
480 
481  int er = ERROR_FAULT;
482 
483  fs_endpoint_t *fsep = (fs_endpoint_t*) ep;
484 
485  // first find the two names separated by "="
486  int p = 0;
487  while (buf[p] != 0 && buf[p] != '=') {
488  p++;
489  }
490  if (buf[p] == 0) {
491  // not found
492  log_error("Did not find '=' in rename command %s\n", buf);
493  return ERROR_SYNTAX_NONAME;
494  }
495 
496  buf[p] = 0;
497  char *from = buf+p+1;
498  char *to = buf;
499 
500  if (index(to, '/') != NULL) {
501  // no separator char
502  log_error("target file name contained dir separator\n");
503  return ERROR_SYNTAX_DIR_SEPARATOR;
504  }
505 
506  char *frompath = malloc_path(fsep->curpath, from);
507  char *topath = malloc_path(fsep->curpath, to);
508 
509  char *fromreal = realpath(frompath, NULL);
510  free(frompath);
511  char *toreal = realpath(topath, NULL);
512 
513  if (toreal != NULL) {
514  // target already exists
515  er = ERROR_FILE_EXISTS;
516  } else {
517  // check both paths against container boundaries
518  if ((strstr(fromreal, fsep->basepath) == fromreal)
519  && (strstr(topath, fsep->basepath) == topath)) {
520  // ok
521 
522  int rv = rename(fromreal, topath);
523 
524  if (rv < 0) {
525  er = errno_to_error(errno);
526  log_errno("Error renaming a file\n");
527  } else {
528  er = ERROR_OK;
529  }
530  }
531  }
532  free(topath);
533  free(toreal);
534  free(fromreal);
535 
536  return er;
537 }

540 static int fs_cd(endpoint_t *ep, char *buf) {
541  fs_endpoint_t *fsep = (fs_endpoint_t*) ep;
542 
543  log_debug("Change dir to: %s\n", buf);
544 
545  // concat new path to current path
546  char *newpath = malloc_path(fsep->curpath, buf);
547 
548  // canonicalize it
549  char *newreal = realpath(newpath, NULL);
550  // free buffer so we don't forget it
551  free(newpath);
552 
553  log_debug("Checking that new path '%s' is under base '%s'\n",
554  newreal, fsep->basepath);
555 
556  if (newreal == NULL) {
557  // target does not exist
558  return ERROR_FILE_NOT_FOUND;
559  }
560 
561  // check if the new path is still under the base path
562  if (strstr(newreal, fsep->basepath) == newreal) {
563  // the needle base path is found at the start of the new real path -> ok
564  free(fsep->curpath);
565  fsep->curpath = newreal;
566  } else {
567  // needle (base path) is not in haystack (new path)
568  // -> security error
569  log_error("Tried to chdir outside base dir %s, to %s\n", fsep->basepath, newreal);
570  free(newreal);
571  return ERROR_NO_PERMISSION;
572  }
573  return ERROR_OK;
574 }
