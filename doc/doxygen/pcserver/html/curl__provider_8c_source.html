119 static curl_endpoint_t *new_endpoint(const char *path) {
120  curl_endpoint_t *fsep = malloc(sizeof(curl_endpoint_t));
121 
122  for (int i = 0; i < MAX_SESSIONS; i++) {
123  File *fp = &(fsep->files[i]);
124  init_fp(fp);
125  }
126 
127  fsep->error_buffer[0] = 0;
128  fsep->name_buffer[0] = 0;
129  fsep->path_buffer[0] = 0;
130  fsep->host_buffer[0] = 0;
131  // separate host from path
132  // if hostend is NULL, then we only have the host
133  char *hostend=index(path, '/');
134 
135  int hostlen = strlen(path);
136  if (hostend != NULL) {
137  hostlen = hostend - path;
138  }
139  strncpy(fsep->host_buffer, path, hostlen);
140  fsep->host_buffer[hostlen] = 0;
141 
142  strncpy(fsep->path_buffer, path+hostlen+1, MAX_BUFFER_SIZE);
143  fsep->path_buffer[MAX_BUFFER_SIZE-1] = 0;
144 
145  return fsep;
146 }

290 static CURLMcode pull_data(curl_endpoint_t *cep, File *fp, int *eof) {
291  int running_handles = 0;
292 
293  *eof = 0;
294  fp->bufdatalen = 0;
295  fp->bufrp = 0;
296 
297  CURLMcode rv = CURLM_OK;
298  do {
299  rv = curl_multi_perform(fp->multi, &running_handles);
300 #ifdef DEBUG_CURL
301  if (rv != 0 || running_handles != 1) {
302  printf("curl read file returns: %d, running=%d\n", rv, running_handles);
303  }
304 #endif
305  } while (rv == CURLM_CALL_MULTI_PERFORM);
306 
307  if (running_handles == 0) {
308  // no more data will be available
309  log_debug("pull_data sets EOF (running=0), rv=%d, datalen=%d\n", rv, fp->bufdatalen);
310  *eof = 1;
311  }
312 
313  int msgs_in_queue = 0;
314  CURLMsg *cmsg = NULL;
315  while ((cmsg = curl_multi_info_read(fp->multi, &msgs_in_queue)) != NULL) {
316 #ifdef DEBUG_CURL
317  printf("cmsg=%p, in queue=%d\n", (void*)cmsg, msgs_in_queue);
318 #endif
319  // we only added one easy handle, so easy_handle is known
320  CURLMSG msg = cmsg->msg;
321  if (msg == CURLMSG_DONE) {
322  CURLcode cc = cmsg->data.result;
323  //printf("cc = %d\n", cc);
324  if (cc != CURLE_OK) {
325  log_error("errorbuffer = %s\n", cep->error_buffer);
326  }
327  log_debug("pull_data sets EOF (err msg)\n");
328  *eof = 1;
329  break;
330  }
331  }
332  return rv;
333 }

335 static int reply_with_data(curl_endpoint_t *cep, File *fp, char *retbuf, int len, int *eof) {
336  // we already have some data to give back
337  int datalen = fp->bufdatalen - fp->bufrp;
338  if (datalen > len) {
339  // we have more data than requested
340  datalen = len;
341  }
342  memcpy(retbuf, &(fp->buffer[fp->bufrp]), datalen);
343  fp->bufrp += datalen;
344 
345  if (fp->bufdatalen <= fp->bufrp) {
346  // reached the end of the buffer
347  CURLMcode rv = CURLM_OK;
348  if (!fp->read_state) { // set from previous pull_data
349  // try to pull in more, so we see if eof should be sent
350  rv = pull_data(cep, fp, &(fp->read_state));
351  if (rv != CURLM_OK) {
352  log_debug("reply_with_data sets EOF rv=%d, datalen=%d\n", rv, fp->bufdatalen);
353  *eof = 1;
354  }
355  }
356 
357  if (fp->read_state && fp->bufdatalen <= fp->bufrp) {
358  log_debug("reply with data sets EOF (2)\n");
359  *eof = 1;
360  }
361  }
362  return datalen;
363 }

366 static int read_file(endpoint_t *ep, int tfd, char *retbuf, int len, int *eof) {
367 
368  curl_endpoint_t *cep = (curl_endpoint_t*) ep;
369  File *fp = find_file(ep, tfd);
370 
371  if (fp != NULL) {
372  if (fp->read_converter != NULL) {
373  // mostly used for directory
374  // I'll never understand where I need a "struct" ...
375  return fp->read_converter((struct curl_endpoint_t*)cep, fp, retbuf, len, eof);
376  }
377 
378  if (fp->bufdatalen > fp->bufrp) {
379 #ifdef DEBUG_CURL
380  printf("Got some data already: datalen=%d\n", fp->bufdatalen);
381 #endif
382  return reply_with_data(cep, fp, retbuf, len, eof);
383  }
384  // we don't have enough data to send something
385  CURLMcode rv = pull_data(cep, fp, &(fp->read_state));
386 
387  if (fp->bufdatalen > fp->bufrp) {
388  // we should now have some data to give back
389 #ifdef DEBUG_CURL
390  printf("Pulled some data: datalen=%d, eof=%d\n", fp->bufdatalen, *eof);
391 #endif
392  return reply_with_data(cep, fp, retbuf, len, eof);
393  }
394 
395  if (rv == CURLM_OK) {
396  return 0;
397  }
398  } else {
399  log_error("fp is NULL on read attempt\n");
400  }
401  return -ERROR_FAULT;
402 }

496 int dir_nlst_read_converter(struct curl_endpoint_t *cep, File *fp, char *retbuf, int len, int *eof) {
497 
498  if (len < FS_DIR_NAME + 1) {
499  log_error("read buffer too small for dir entry (is %d, need at least %d)\n",
500  len, FS_DIR_NAME+1);
501  return -ERROR_FAULT;
502  }
503 
504  // prepare dir entry
505  memset(retbuf, 0, FS_DIR_NAME+1);
506 
507  int l = 0;
508  char *namep = retbuf + FS_DIR_NAME;
509  switch(fp->read_state) {
510  case 0: // disk name
511  retbuf[FS_DIR_MODE] = FS_DIR_MOD_NAM;
512  l = strlen(fp->name_buffer);
513  if (len < FS_DIR_NAME + l + 1) {
514  log_error("read buffer too small for dir name (is %d, need at least %d)\n",
515  len, l+FS_DIR_NAME+1);
516  return -ERROR_FILE_NAME_TOO_LONG;
517  }
518  strncpy(retbuf, fp->name_buffer, l+1);
519  l = l + FS_DIR_NAME + 1;
520  fp->read_state++;
521  break;
522  case 1:
523  // file names
524 #ifdef DEBUG_CURL
525  log_debug("get filename, bufdatalen=%d, bufrp=%d, eof=%d\n",
526  fp->bufdatalen, fp->bufrp, *eof);
527 #endif
528 
529  l = FS_DIR_NAME;
530  retbuf[FS_DIR_MODE] = FS_DIR_MOD_FIL;
531  do {
532  while ((fp->bufdatalen <= fp->bufrp) && (*eof == 0)) {
533 
534  //log_debug("Trying to pull...\n");
535 
536  CURLMcode rv = pull_data((curl_endpoint_t*)cep, fp, eof);
537 
538  if (rv != CURLM_OK) {
539  log_error("Error retrieving directory data (%d)\n", rv);
540  return -ERROR_DIR_ERROR;
541  }
542  }
543  // find length of name
544 
545  while (fp->bufrp < fp->bufdatalen) {
546  char c = fp->buffer[fp->bufrp];
547  fp->bufrp ++;
548  if (c == 13) {
549  // ignore
550  } else
551  if (c != 10) {
552  *namep = c;
553  namep++;
554  l++;
555  } else {
556  // end of name
557  *namep = 0;
558  l++;
559 #ifdef DEBUG_CURL
560  log_debug("bufrp=%d, datalen=%d\n", fp->bufrp, fp->bufdatalen);
561 #endif
562  break;
563  }
564  if (len < FS_DIR_NAME + l + 1) {
565  log_error("read buffer too small for dir name (is %d, need at least %d)\n",
566  len, l+FS_DIR_NAME+1);
567  return -ERROR_FILE_NAME_TOO_LONG;
568  }
569  }
570  if (*eof != 0) {
571  log_debug("end of dir read\n");
572  fp->read_state++;
573  *namep = 0;
574  }
575  }
576  while (*namep != 0 && *eof == 0); // not null byte, then not done
577  *eof = 0;
578  if (l > FS_DIR_NAME) {
579  break;
580  }
581  // otherwise fall through
582  case 2:
583  log_debug("final dir entry\n");
584  retbuf[FS_DIR_MODE] = FS_DIR_MOD_FRE;
585  retbuf[FS_DIR_NAME] = 0;
586  l = FS_DIR_NAME + 1;
587  *eof = 1;
588  fp->read_state++;
589  break;
590  }
591 #ifdef DEBUG_CURL
592  printf("Got some dir data : datalen=%d\n", l);
593  for (int i = 0; i < FS_DIR_NAME; i++) {
594  printf("%02x ", retbuf[i]);
595  }
596  printf(": %s\n", retbuf+FS_DIR_NAME);
597 #endif
598  return l;
599 }

661

663 
664  log_debug("CHDIR: Current path is '%s', name is '%s'\n", cep->path_buffer, name);

671  strncpy(cep->path_buffer, name, MAX_BUFFER_SIZE);
672  cep->path_buffer[MAX_BUFFER_SIZE-1] = 0;
673  return 0;
674  }
