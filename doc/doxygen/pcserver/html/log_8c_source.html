30 void log_term(const char *msg) {
31 
32  int newline = 0;
33 
34  printf(">>>: ");
35  char c;
36  while ((c = *msg) != 0) {
37  if (newline) {
38  printf("\n>>>: ");
39  }
40  if (isprint(c)) {
41  putchar(c);
42  newline = 0;
43  } else
44  if (c == 10) {
45  newline = 1;
46  } else
47  if (c == 13) {
48  // silently ignore
49  } else {
50  printf("{%02x}", ((int)c) & 0xff);
51  newline = 0;
52  }
53  msg++;
54  }
55  printf("\n");
56 }
