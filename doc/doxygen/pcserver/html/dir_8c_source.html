103 int dir_call_matches(const char *dir, const char *pattern, int (*callback)(const int num_of_match, const char *name)) {
104  int matches = 0;
105  DIR *dp;
106  int rv;
107  struct dirent *de;
108  int onlyone = 0;
109 
110  // shortcut - if we don't have wildcards, just open it
111  if (index(pattern, '*') == NULL && index(pattern, '?') == NULL) {
112  onlyone = 1;
113  }
114 
115 
116  dp = opendir(dir);
117  if (dp) {
118  de = readdir(dp);
119 
120  while (de != NULL) {
121  if (compare_pattern(de->d_name, pattern)) {
122  // match
123  matches ++;
124 
125  // get full path
126  char *namebuf = malloc_path(dir, de->d_name);
127  rv = callback(matches, namebuf);
128  // free memory
129  free(namebuf);
130  if (rv || onlyone) {
131  // either callback tells us to stop
132  // or there are no wildcards, so this has to be it
133  closedir(dp);
134  // if rv < 0 then some kind of error happened, return it
135  // instead of the number of matches
136  return (rv < 0) ? rv : matches;
137  }
138  }
139  de = readdir(dp);
140  }
141 
142  }
143  closedir(dp);
144  return matches;
145 }

203 int dir_fill_entry(char *dest, char *curpath, struct dirent *de, int maxsize) {
204  struct stat sbuf;
205  struct tm *tp;
206 
207  char *realname = malloc_path(curpath, de->d_name);
208 
209  /* TODO: check return value */
210  int rv = stat(realname, &sbuf);
211  if (rv < 0) {
212  log_error("Failed stat'ing entry %s\n", de->d_name);
213  log_errno("Problem stat'ing dir entry");
214  }
215  free(realname);
216 
217  dest[FS_DIR_LEN] = sbuf.st_size & 255;
218  dest[FS_DIR_LEN+1] = (sbuf.st_size >> 8) & 255;
219  dest[FS_DIR_LEN+2] = (sbuf.st_size >> 16) & 255;
220  dest[FS_DIR_LEN+3] = (sbuf.st_size >> 24) & 255;
221 
222  tp = localtime(&sbuf.st_mtime);
223  dest[FS_DIR_YEAR] = tp->tm_year;
224  dest[FS_DIR_MONTH] = tp->tm_mon;
225  dest[FS_DIR_DAY] = tp->tm_mday;
226  dest[FS_DIR_HOUR] = tp->tm_hour;
227  dest[FS_DIR_MIN] = tp->tm_min;
228  dest[FS_DIR_SEC] = tp->tm_sec;
229 
230  dest[FS_DIR_MODE] = S_ISDIR(sbuf.st_mode) ? FS_DIR_MOD_DIR : FS_DIR_MOD_FIL;
231  // de->d_name is 0-terminated (see readdir man page)
232  int l = strlen(de->d_name);
233  strncpy(dest+FS_DIR_NAME, de->d_name,
234  min(l+1, maxsize-1-FS_DIR_NAME));
235  // make sure we're still null-terminated
236  dest[maxsize-1] = 0;
237 
238  return FS_DIR_NAME + strlen(dest+FS_DIR_NAME) + 1;
239 }
