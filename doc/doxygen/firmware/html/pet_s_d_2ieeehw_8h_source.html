73 static inline void ndachi (void) {
74  // disable interrupt to avoid race condition
75  // of ATN irq between the atnishi() check and
76  // setting NDAC lo
77  cli();
78  if (atnishi() || is_atna) {
79  IEEE_PORT_NDAC |= _BV(IEEE_PIN_NDAC); // NDAC high

206 // switch hardware from receive to transmit (to talk)
207 static inline void settx (void) {
208  IEEE_DDR_NDAC &= (uint8_t)~_BV(IEEE_PIN_NDAC); // NDAC as input
209  IEEE_DDR_NRFD &= (uint8_t)~_BV(IEEE_PIN_NRFD); // NRFD as input
210  IEEE_PORT_NDAC |= _BV(IEEE_PIN_NDAC); // Enable NDAC pull-up
211  IEEE_PORT_NRFD |= _BV(IEEE_PIN_NRFD); // Enable NRFD pull-up
212  davhi(); // Release DAV
213  eoihi(); // EOI high
214  IEEE_D_PORT = 0xff; // Release data lines
215  tehi(); // Bus driver => TALK
216  IEEE_DDR_DAV |= _BV(IEEE_PIN_DAV); // DAV as output (high)
217  IEEE_D_DDR = 0xff; // Data as output (high)
218  /* We're faster than the PET, so we have to wait for NDAC low first
219  * to avoid running in DEVICE NOT PRESENT error */
220  while(ndacishi() && atnishi()); // Wait for NDAC low
221 }
222 
223 // switch hardware from transmit to receive (after talk)
224 // this happens after ATN, so nrfd and ndac are already low
225 static inline void setrx (void) {
226  IEEE_DDR_DAV &= (uint8_t)~_BV(IEEE_PIN_DAV); // DAV as input
227  IEEE_D_DDR = 0; // Data as input
228  IEEE_PORT_DAV |= _BV(IEEE_PIN_DAV); // Enable DAV pull-up
229  IEEE_D_PORT = 0xff; // Enable data pull-ups
230  eoihi(); // Release EOI
231  nrfdhi(); // Release NRFD
232  ndachi(); // Release NDAC
233  telo(); // Bus driver => LISTEN
234  IEEE_DDR_NDAC |= _BV(IEEE_PIN_NDAC); // NDAC as output
235  IEEE_DDR_NRFD |= (uint8_t) _BV(IEEE_PIN_NRFD); // NRFD as output
236 }
