166 static int8_t directory_converter(packet_t *p) {
167  uint8_t *inp = NULL;
168  uint8_t *outp = &(out[0]);
169 
170  if (p == NULL) {
171  debug_puts("P IS NULL!");
172  return -1;
173  }
174 
175  inp = packet_get_buffer(p);
176  uint8_t type = inp[FS_DIR_MODE];
177 
178  //packet_update_wp(p, 2);
179 
180  if (type == FS_DIR_MOD_NAM) {
181  *outp = 1; outp++; // load address low
182  *outp = 4; outp++; // load address high
183  }
184 
185  *outp = 1; outp++; // link address low; will be overwritten on LOAD
186  *outp = 1; outp++; // link address high
187 
188  uint16_t lineno = 0;
189  // line number, derived from file length
190  if (inp[FS_DIR_LEN+3] != 0
191  || inp[FS_DIR_LEN+2] > 0xf9
192  || (inp[FS_DIR_LEN+2] == 0xf9 && inp[FS_DIR_LEN+1] == 0xff && inp[FS_DIR_LEN] != 0)) {
193  // more than limit of 63999 blocks
194  lineno = 63999;
195  } else {
196  lineno = inp[FS_DIR_LEN+1] | (inp[FS_DIR_LEN+2] << 8);
197  if (inp[FS_DIR_LEN] != 0) {
198  lineno++;
199  }
200  }
201  *outp = lineno & 255; outp++;
202  *outp = (lineno>>8) & 255; outp++;
203 
204  //snprintf(outp, 5, "%hd", (unsigned short)lineno);
205  //outp++;
206  if (lineno < 10) { *outp = ' '; outp++; }
207  if (type == FS_DIR_MOD_NAM) {
208  *outp = 0x12; // reverse for disk name
209  outp++;
210  } else {
211  if (type != FS_DIR_MOD_FRE) {
212  if (lineno < 100) { *outp = ' '; outp++; }
213  if (lineno < 1000) { *outp = ' '; outp++; }
214  if (lineno < 10000) { *outp = ' '; outp++; }
215  }
216  }
217 
218  if (type != FS_DIR_MOD_FRE) {
219  *outp = '"'; outp++;
220  uint8_t i = FS_DIR_NAME;
221  // note the check i<16 - this is buffer overflow protection
222  // file names longer than 16 chars are not displayed
223  while ((inp[i] != 0) && (i < (FS_DIR_NAME + 16))) {
224  *outp = ascii_to_petscii(inp[i]);
225  outp++;
226  i++;
227  }
228  // note: not counted in i
229  *outp = '"'; outp++;
230 
231  // fill up with spaces, at least one space behind filename
232  while (i < FS_DIR_NAME + 16 + 1) {
233  *outp = ' '; outp++;
234  i++;
235  }
236  }
237 
238  // add file type
239  if (type == FS_DIR_MOD_NAM) {
240  // file name entry
241  outp = append(outp, SW_NAME_LOWER);
242  //strcpy(outp, SW_NAME_LOWER);
243  //outp += strlen(SW_NAME_LOWER)+1; // includes ending 0-byte
244  } else
245  if (type == FS_DIR_MOD_DIR) {
246  outp = append(outp, "dir");
247  //strcpy(outp, "dir");
248  //outp += 4; // includes ending 0-byte
249  } else
250  if (type == FS_DIR_MOD_FIL) {
251  outp = append(outp, "prg");
252  //strcpy(outp, "prg");
253  //outp += 4; // includes ending 0-byte
254  } else
255  if (type == FS_DIR_MOD_FRE) {
256  outp = append(outp, "blocks free");
257  //strcpy(outp, "bytes free");
258  //outp += 11; // includes ending 0-byte
259 
260  *outp = 0; outp++; // BASIC end marker (zero link address)
261  *outp = 0; outp++; // BASIC end marker (zero link address)
262  }
263 
264  uint8_t len = outp - out;
265  if (len > packet_get_capacity(p)) {
266  debug_puts("CONVERSION NOT POSSIBLE!"); debug_puthex(len); debug_putcrlf();
267  return -1; // conversion not possible
268  }
269 
270 #if DEBUG_SERIAL
271  debug_puts("CONVERTED TO: LEN="); debug_puthex(len);
272  for (uint8_t j = 0; j < len; j++) {
273  debug_putc(' '); debug_puthex(out[j]);
274  }
275  debug_putcrlf();
276 #endif
277  // this should probably be combined
278  memcpy(packet_get_buffer(p), &out, len);
279  packet_update_wp(p, len);
280 
281  return 0;
282 }

306 // returns the next byte to send - first the packet type, then the data length
307 // then the channel and then the data (len bytes).
308 //
309 // Used by UART code to include type and length in packet
310 //
311 // returns -1 if empty
312 static int16_t read_char_from_packet(packet_t *buf) {
313  int16_t rv = -1;
314  switch (txstate) {
315  case TX_TYPE:
316  txstate = TX_LEN;
317  rv = 0xff & packet_get_type(buf);
318  break;
319  case TX_LEN:
320  txstate = TX_CHANNEL;
321  rv = (0xff & packet_get_contentlen(buf)) + 3; // plus 2 to adhere to packet wire format
322  break;
323  case TX_CHANNEL:
324  rv = 0xff & packet_get_chan(buf);
325  if (packet_get_contentlen(buf) == 0) {
326  txstate = TX_IDLE;
327  } else {
328  txstate = TX_DATA;
329  }
330  break;
331  case TX_DATA:
332  rv = 0xff & packet_read_char(buf);
333  if (!packet_has_data(buf)) {
334  txstate = TX_IDLE;
335  }
336  break;
337  default:
338  break;
339  }
340  return rv;
341 }

377 static void push_data_to_packet(int8_t rxdata)
378 {
379  switch(rxstate) {
380  case RX_IDLE:
381  // no current packet
382  if (rxdata == FS_SYNC) {
383  // sync received
384  // mirror the sync back
385  // first flush all packets
386  while (slots_used > 0) {
387  send();
388  }
389  // can we send?
390  while (!uarthw_can_send());
391  // yes, send sync
392  uarthw_send(FS_SYNC);
393  } else
394  if (rxdata == FS_REPLY || rxdata == FS_WRITE || rxdata == FS_EOF) {
395  // note EOI flag
396  current_is_eoi = rxdata;
397  // start a reply handling
398  rxstate = RX_LEN;
399  } else {
400  //led_toggle();
401  }
402  break;
403  case RX_LEN:
404  current_data_left = rxdata - 3;
405  rxstate = RX_CHANNELNO;
406  break;
407  case RX_CHANNELNO:
408  current_channelno = rxdata;
409  rxstate = RX_IGNORE; // fallback
410  // find the current receive buffer
411  for (uint8_t i = 0; i < NUMBER_OF_SLOTS; i++) {
412  if (rx_channels[i].channelno == current_channelno) {
413 //if (current_is_eoi == FS_EOF && current_data_left == 0) led_toggle();
414  current_channelpos = i;
415  current_rxpacket = rx_channels[current_channelpos].rxpacket;
416  if (packet_set_write(current_rxpacket, current_channelno,
417  current_is_eoi, current_data_left) >= 0) {
418  rxstate = RX_DATA;
419  }
420  break;
421  }
422  }
423  // well, RX_IGNORE should not happen, but we have no means of telling anyone here
424  if (current_data_left == 0) {
425  // we are actually already done. do callback and set status to idle
426  rx_channels[current_channelpos].callback(current_channelno,
427  (rxstate == RX_IGNORE) ? -1 : 0);
428  rx_channels[current_channelpos].channelno = -1;
429  rxstate = RX_IDLE;
430  }
431  break;
432  case RX_DATA:
433  packet_write_char(current_rxpacket, rxdata);
434  // fallthrough
435  case RX_IGNORE:
436  current_data_left --;
437 //if (packet_get_contentlen(current_rxpacket) > 1) led_on();
438  if (current_data_left <= 0) {
439  rx_channels[current_channelpos].callback(current_channelno,
440  (rxstate == RX_IGNORE) ? -1 : 0);
441  rx_channels[current_channelpos].channelno = -1;
442  rxstate = RX_IDLE;
443  }
444  break;
445  default:
446  break;
447  }
448 }

484 void serial_submit(void *epdata, packet_t *buf) {
485 
486  // wait for slot free
487  while (slots_used >= (NUMBER_OF_SLOTS-1)) {
488  serial_delay();
489  }
490 
491 
492  // protect slot* by disabling the interrupt
493  //cli(); - as slots are not handled in irq anymore
494  // note: slots_used can only decrease until here, as this is the
495  // only place to increase it, so there is no race from the while()
496  // above to setting it here.
497  slots[slots_used] = buf;
498  slots_used++;
499  if (slots_used == 1) {
500  // no packet before, so need to start sending
501  txstate = TX_TYPE;
502  send();
503  //led_on();
504  }
505  // enable interrupts again
506  //sei(); - as slots are not handled in irq anymore
507 }

509 /*****************************************************************************
510  * submit a channel rpc call to the UART
511  * If buffer slot is available, return immediately.
512  * Otherwise wait until slot is available, then return
513  * (while data is transferred in the background)
514  *
515  * Note: submitter must check buf_is_empty() for true or buf_wait_free()
516  * before reuse or freeing the memory!
517  *
518  * callback is called from interrupt context when the response has been
519  * received
520  */
521 void serial_submit_call(void *epdata, int8_t channelno, packet_t *txbuf, packet_t *rxbuf,
522  void (*callback)(int8_t channelno, int8_t errnum)) {
523 
524  if (channelno < 0) {
525  debug_printf("!!!! submit with channelno=%d\n", channelno);
526  }
527  if (txbuf->chan < 0) {
528  debug_printf("!!!! submit with packet->chan=%d\n", txbuf->chan);
529  }
530 
531  // check rx slot
532  // wait / loop until receive buffer is being freed by interrupt routine
533  int8_t channelpos = -1;
534  while (channelpos < 0) {
535  for (uint8_t i = 0; i < NUMBER_OF_SLOTS; i++) {
536  if (rx_channels[i].channelno < 0) {
537  channelpos = i;
538  break;
539  }
540  }
541  serial_delay();
542  }
543 
544  rx_channels[channelpos].channelno = channelno;
545  rx_channels[channelpos].rxpacket = rxbuf;
546  rx_channels[channelpos].callback = callback;
547 
548  // send request
549  serial_submit(epdata, txbuf);
550 }
