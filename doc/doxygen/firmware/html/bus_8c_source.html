120 static void _cmd_callback(int8_t errnum, uint8_t *rxdata) {
121  bus_for_irq->errnum = errnum;
122  if (errnum == 1) {
123  if (rxdata != NULL) {
124  bus_for_irq->errparam = rxdata[1];

132 static int16_t cmd_handler (bus_t *bus)
133 {
134 
135  int16_t st = 0;
136 
137  bus->cmd_done = 0;
138 
139  uint8_t secaddr = bus->secondary & 0x0f;
140 
141  int8_t rv = 0;
142 
143  // prepare for callback from interrupt
144  bus_for_irq = bus;
145  bus_for_irq->cmd_done = 0;
146 
147  // zero termination
148  bus->command.command_buffer[bus->command.command_length++] = 0;
149 
150  if (secaddr == 0x0f) {
151  /* Handle commands */
152  // zero termination
153  rv = command_execute(bus_secaddr_adjust(bus, secaddr), bus, &error, _cmd_callback);
154 
155  // change device address after command
156  bus_for_irq->current_device_address = bus_for_irq->rtconf.device_address;
157  } else {
158  /* Handle filenames */
159 
160 #ifdef DEBUG_SERIAL
161  debug_printf("Open file secaddr=%02x, name='%s'\n",
162  secaddr, bus->command.command_buffer);
163 #endif
164  rv = file_open(bus_secaddr_adjust(bus, secaddr), bus, &error,
165  _cmd_callback, secaddr == 1);
166  }
167 
168  if (rv < 0) {
169  // open ran into an error
170  // -- errormsg should be already set, so nothing left to do here
171  // TODO
172  debug_printf("Received direct error number on open: %d\n", rv);
173  set_error(&error, ERROR_DRIVE_NOT_READY);
174  st = 2;
175  } else {
176  // as this code is not (yet?) prepared for async operation, we
177  // need to wait here until the response from the server comes
178  // Note: use bus_for_irq here, as it is volatile
179  while (bus_for_irq->cmd_done == 0) {
180  // TODO this should be reworked more backend (serial) independent
181  serial_delay();
182  }
183 #ifdef DEBUG_SERIAL
184  debug_printf("Received callback error number on open: %d\n", bus_for_irq->errnum);
185 #endif
186  // result of the open
187  if (bus_for_irq->errnum == 1) {
188  set_error_ts(&error, bus_for_irq->errnum, bus_for_irq->errparam, 0);
189  } else {
190  set_error(&error, bus_for_irq->errnum);
191  }
192 
193  // command may (or may not) open channel 15 for callback to the server
194  // so close it here, as this is done separately
195  if (secaddr == 15) {
196  channel_close(bus_secaddr_adjust(bus, 15));
197  }
198 
199  if (bus_for_irq->errnum != 0) {
200  channel_close(bus_secaddr_adjust(bus, secaddr));
201  } else {
202  // really only does something on read-only channels
203  channel_preload(bus_secaddr_adjust(bus, secaddr));
204  }
205  }
206 
207  bus_for_irq->command.command_length = 0;
208 
209  return st;
210 }

217 // called during listenloop to send bytes to the server
218 int16_t bus_sendbyte(bus_t *bus, uint8_t data, uint8_t with_eoi) {
219 
220  int16_t st = 0;
221 #ifdef DEBUG_SERIAL_DATA
222  debug_printf("sendbyte: %02x (%c)\n", data, (isprint(data) ? data : '-'));
223 #endif
224 
225  if((bus->secondary & 0x0f) == 0x0f || (bus->secondary & 0xf0) == 0xf0) {
226  if (bus->command.command_length < CONFIG_COMMAND_BUFFER_SIZE) {
227  bus->command.command_buffer[bus->command.command_length++] = data;
228  }
229  } else {
230  bus->channel = channel_put(bus->channel, data, with_eoi);
231  if (bus->channel == NULL) {
232  st = 0x83; // TODO correct code
233  }
234  }
235  return st + (bus->device << 8);
236 }

239 // called during talkloop to receive bytes from the server
240 // If the preload parameter is set, the data byte is set,
241 // but the read pointers are not advanced (used to be named "fake"...)
242 int16_t bus_receivebyte(bus_t *bus, uint8_t *data, uint8_t preload) {
243 
244  int16_t st = 0;
245 
246  uint8_t secaddr = bus->secondary & 0x0f;
247  channel_t *channel = bus->channel;
248 
249  if (secaddr == 15) {
250 
251  *data = error.error_buffer[error.readp];
252 
253  if (error.error_buffer[error.readp+1] == 0) {
254  // send EOF
255  st |= 0x40;
256  }
257  if (!preload) {
258  // the real thing
259  error.readp++;
260  if (error.error_buffer[error.readp] == 0) {
261  // finished reading the error message
262  // set OK
263  set_error(&error, ERROR_OK);
264  }
265  }
266  } else {
267  if (channel == NULL) {
268  // if still NULL, error
269  debug_printf("Setting file not open on secaddr %d\n", bus->secondary);
270 
271  set_error(&error, ERROR_FILE_NOT_OPEN);
272  st = 0x83;
273  } else {
274 #ifdef DEBUG_SERIAL
275  //debug_printf("rx: chan=%p, channo=%d\n", channel, channel->channel_no);
276 #endif
277  // first fillup the buffers
278  channel_preloadp(channel);
279 
280  *data = channel_current_byte(channel);
281  if (channel_current_is_eof(channel)) {
282 #ifdef DEBUG_SERIAL
283  debug_puts("EOF!\n");
284 #endif
285  st |= 0x40;
286  }
287 
288  if (!preload) {
289  // make sure the next call does have a data byte
290  if (!channel_next(channel)) {
291  if (channel_has_more(channel)) {
292  channel_refill(channel);
293  } else {
294  if (secaddr == 15 || secaddr == 0) {
295  // autoclose when load is done, or after reading status channel
296  channel_close(bus_secaddr_adjust(bus, secaddr));
297  bus->channel = NULL;
298  }
299  }
300  }
301  }
302  }
303  }
304  return st + (bus->device << 8);
305 }

308 static int16_t bus_prepare(bus_t *bus)
309 {
310  uint8_t b;
311  int8_t secaddr;
312  int st = 0;
313 
314 #ifdef DEBUG_SERIAL
315  debug_printf("***ParallelCommand %02x %02x\n",
316  bus->device, bus->secondary);
317 #endif
318 
319  if ((bus->device & 0x0f) != bus->current_device_address) {
320  return 0x80; // device not present
321  }
322 
323  secaddr = bus->secondary & 0x0f;
324 
325  if (secaddr != 15) {
326  /* Open Channel */
327  bus->channel = channel_find(bus_secaddr_adjust(bus, secaddr));
328  if (bus->channel == NULL) {
329  debug_puts("Did not find channel!\n");
330  st |= 0x40; // TODO correct code?
331  }
332  }
333  if ((bus->device & 0xf0) == 0x40) {
334  // if we should TALK, prepare the first data byte
335  if (!st) {
336  st = bus_receivebyte(bus, &b, 1) & 0xbf; /* any error, except eof */
337  //debug_printf("receive gets st=%04x\n", st);
338  } else {
339  // cause a FILE NOT FOUND
340  debug_puts("FILE NOT FOUND\n");
341  bus->device = 0;
342  }
343  }
344  return 0;
345 }

359 int16_t bus_attention(bus_t *bus, uint8_t b) {
360  int16_t st = 0;
361 
362  // UNLISTEN and it is either open or the command channel
363  if (b == 0x3f
364  && (((bus->secondary & 0xf0) == 0xf0)
365  || ((bus->secondary & 0x0f) == 0x0f))) {
366 
367  if ((bus->device & 0x0f) == bus->current_device_address) {
368  // then process the command
369  st = cmd_handler(bus);
370  }
371 
372  } else {
373 
374  // not open, not command:
375  switch (b & 0xf0) {
376  case 0x20:
377  case 0x40:
378  // store device number plus LISTEN/TALK info
379  if ((b & 0x0f) == bus->current_device_address) {
380  bus->device = b;
381  }
382  break;
383 
384  case 0x60:
385  // secondary address (open DATA channel)
386  if ((bus->device & 0x0f) == bus->current_device_address) {
387 
388  bus->secondary = b;
389  // process a command if necessary
390  bus_prepare(bus);
391  }
392  break;
393  case 0xe0:
394  // secondary address (CLOSE)
395  if ((bus->device & 0x0f) == bus->current_device_address) {
396  bus->secondary = b;
397  // process a command if necessary
398  //st = bus_command(bus);
399  bus_close(bus);
400  }
401  break;
402 
403  case 0xf0: /* Open File needs the filename first */
404  if ((bus->device & 0x0f) == bus->current_device_address) {
405  bus->secondary = b;
406  // TODO: close previously opened file
407  }
408  break;
409  default:
410  // falls through e.g. for unlisten/untalk
411  break;
412  }
413  }
414 
415  if ((b == 0x3f) || (b == 0x5f) || ((b & 0xf0) == 0xe0)) {
416  // unlisten, untalk, close
417  bus->device = 0;
418  bus->secondary = 0;
419 
420  led_set(IDLE);
421  } else {
422  if (bus->current_device_address != (bus->device & 0x0f)) {
423  // not this device
424  st |= 0x80;
425  } else {
426  led_set(ACTIVE);
427  }
428  }
429 
430 #ifdef DEBUG_SERIAL
431  debug_printf("ParallelAttention(%02x)->TrapDevice=%02x, st=%04x\n",
432  b, bus->device, st | (bus->device << 8));
433  debug_flush();
434 #endif
435 
436  st |= bus->device << 8;
437 
438  return st;
439 }
