60 // read a byte from IEEE
61 int16_t liecin(int *c)
62 {
63  int er = 0;
64 
65  nrfdhi();
66 
67  // do...while to make sure to at least test ATN once
68  do {
69  if(atnislo()) {
70  // ATN got low, exit
71  goto atn;
72  }
73  } while( davishi() );
74 
75  nrfdlo();
76 
77  if(!eoiishi())
78  er|=E_EOI;
79  *c=rdd();
80 
81  ndachi();
82 
83  do {
84  if(atnislo()) {
85  // ATN got low, exit
86  //break;
87  goto atn;
88  }
89  } while( davislo() );
90 
91  ndaclo();
92 
93  return(er);
94 
95 atn:
96  // we do not need to do anything,
97  // as we already are in listen mode, and the
98  // ACK hardware or software has just pulled
99  // NRFD and NDAC
100  return E_ATN;
101 }

103 void listenloop() {
104 #ifdef DEBUG_BUS
105  debug_putc('L');
106 #endif
107  int er, c;
108  while(((er=liecin(&c))&E_ATN)!=E_ATN) {
109  par_status = bus_sendbyte(&bus, c, er & E_EOI);

121 void talkloop()
122 {
123  int16_t er /*,sec*/;
124  uint8_t c;
125 
126 #ifdef DEBUG_BUS
127  debug_putc('T'); debug_flush();
128 #endif
129  settx(); /* enables sending */
130 
131  er=0;
132  /*sec=secadr&0x0f;*/
133 
134  while (!er) {
135 
136  /* wait nrfd hi */
137  do {
138  if( atnislo() ) {
139  goto atn;
140  }
141  } while( nrfdislo() );
142 
143  /* write data & eoi */
144  par_status = bus_receivebyte(&bus, &c, 1);
145  if(par_status & 0x40)
146  {
147  eoilo();
148  er|=E_EOI;
149  }
150  wrd(c);
151  davlo();
152 
153  /* wait nrfd lo */
154  do {
155  if( atnislo() ) {
156  goto atn;
157  }
158 
159  if( ndacishi() && nrfdishi() ) {
160  eoihi();
161  davhi();
162  er |= E_NODEV;
163  goto idle;
164  }
165  } while( nrfdishi() );
166 
167  /* wait ndac hi */
168  do {
169  if ( atnislo() ) {
170  // ATN got low
171  goto atn;
172  }
173  } while( ndacislo() );
174 
175  davhi();
176  eoihi();
177  par_status = bus_receivebyte(&bus, &c, 0);
178 
179  /* wait ndac lo */
180  if( par_status & 0xff ) {
181  break;
182  } else {
183  do {
184  if ( atnislo() ) {
185  goto atn;
186  }
187  } while( ndacishi() );
188  }
189  }
190  // no ATN, so set bus to idle
191  setidle();
192  return; //(er&(E_EOI));
193 
194 atn:
195  // sets IEEE488 back to receive mode
196  setrx();
197  return; //(er&(E_EOI));
198 
199 idle:
200  // after EOF we set bus to idle
201  setidle();
202  return; //(er&(E_EOI));
203 }

211 void ieee_mainloop_iteration(void)
212 {
213  int cmd = 0;
214 
215  // only do something on ATN low
216  if (atnishi()) {
217  return;
218  }
219 
220  // set receive mode
221  setrx();
222 
223  par_status=0;
224 
225  /* Loop to get commands during ATN lo ----------------------------*/
226 
227  while(1)
228  {
229  ndaclo();
230  // acknowledge ATN
231  atnalo();
232  nrfdhi();
233 
234  /* wait for DAV lo */

249  par_status = bus_attention(&bus, cmd);
