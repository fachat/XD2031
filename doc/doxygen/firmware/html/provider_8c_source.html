74 int8_t provider_assign(uint8_t drive, const char *name) {
75 
76  int8_t rv = -1;
77 
78 #ifdef DEBUG_PROVIDER
79  debug_printf("ASSIGN: drive %d to '%s'\n", drive, name);
80 #endif
81 
82  // check drive
83  if (drive > 10) {
84  return -1;
85  }
86 
87  // first find the colon
88  uint8_t p = 0;
89  while (name[p] != 0 && name[p] != ':') {
90  p++;
91  }
92 
93  provider_t *newprov = NULL;
94  void *provdata = NULL;
95 
96  if ((isdigit(name[0])) && (p == 1)) {
97  // we have a drive digit
98  uint8_t drv = name[0] & 0x0f;
99  endpoint_t *ep = provider_lookup(drv);
100  if (ep == NULL) {
101  debug_printf("Drive %d not used, cannot do relative assign!\n", drv);
102  return -1;
103  }
104  if (ep == &default_provider) {
105  debug_puts("Default is not registered, forwarding to server\n");
106  return -1;
107  }
108  newprov = ep->provider;
109  // currently a relative assignment is not supported
110  provdata = newprov->prov_assign(name);
111  }
112 
113  if (newprov == NULL) {
114  // now check each provider in turn, if the name fits
115  for (int8_t i = MAX_PROV-1; i >= 0; i--) {
116  uint8_t j;
117  if (provs[i].name != NULL) {
118  //debug_printf("Compare with %s\n",provs[i].name);
119  for (j = 0; j < p; j++) {
120  if (provs[i].name[j] != name[j]) {
121  // name does not match
122  //debug_printf("name does not match at i=%d, j=%d (%02x vs. %02x\n",
123  // i, j, provs[i].name[j], name[j]);
124  break;
125  }
126  }
127  //debug_printf("j=%d, p=%d, %02x\n", j, p, provs[i].name[p]);
128  if ((j == p) && (provs[i].name[p] == 0)) {
129  // found it
130  //debug_printf("Found new provider: %p in slot %d\n", p, i);
131  newprov = provs[i].provider;
132  // new get the runtime data
133  provdata = newprov->prov_assign(name);
134  break;
135  }
136  }
137  }
138  }
139 
140  // when we are going to return, remove
141  // the old assign mapping
142  for (int8_t i = MAX_DRIVES-1; i >= 0; i--) {
143  if (drives[i].drive == drive) {
144  drives[i].drive = -1;
145  drives[i].endpoint.provider = NULL;
146  break;
147  }
148  }
149 
150  if (newprov != NULL) {
151  // got one, so register it
152  for (int8_t i = MAX_DRIVES-1; i >= 0; i--) {
153  if (drives[i].drive == -1) {
154  drives[i].drive = drive;
155  drives[i].endpoint.provider = newprov;
156  drives[i].endpoint.provdata = provdata;
157 #ifdef DEBUG_PROVIDER
158  debug_printf("Register prov %p for drive %d with data %p in slot %d\n",
159  newprov, drive, provdata, i);
160 #endif
161  rv = 0;
162  break;
163  }
164  }
165  }
166 
167  return rv;
168 }

170 endpoint_t* provider_lookup(uint8_t drive) {
171 
172 #ifdef DEBUG_PROVIDER
173  debug_printf("provider_lookup for drive %d\n", drive);
174 #endif
175  for (int8_t i = MAX_DRIVES-1; i >= 0; i--) {
176  if ((drives[i].drive == drive) && (drives[i].endpoint.provider != NULL)) {
177 #ifdef DEBUG_PROVIDER
178  debug_printf("found %p in slot %d (default=%p)\n",
179  drives[i].endpoint.provider, i, default_provider.provider);
180 #endif
181  return &(drives[i].endpoint);
182  }
183  }
184 #ifdef DEBUG_PROVIDER
185  debug_puts("not found, returning default\n");
186 #endif
187  return &default_provider;
188 }

190 uint8_t provider_register(const char *name, provider_t *provider) {
191 
192 #ifdef DEBUG_PROVIDER
193  debug_printf("Register provider %p for '%s'\n", provider, name);
194 #endif
195 
196  for (int8_t i = MAX_PROV-1; i >= 0; i--) {
197  if (provs[i].name == NULL) {
198  provs[i].name = name;
199  provs[i].provider = provider;
200  break;
201  }
202  }
203  return 0;
204 }
