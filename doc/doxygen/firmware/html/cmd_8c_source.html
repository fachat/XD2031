35 command_t command_find(uint8_t *input) {
36  switch(*input) {
37  case '$':
38  return CMD_DIR;
39  break;
40  case 'I':
41  return CMD_INITIALIZE;
42  break;
43  case 'R':
44  if (*(input+1) == 'M' || *(input+1) == 'D') {
45  // RMDIR or RD
46  return CMD_RMDIR;
47  }
48  return CMD_RENAME;
49  break;
50  case 'S':
51  return CMD_SCRATCH;
52  break;
53  case 'C':
54  if (*(input+1) == 'D' || *(input+1) == 'H') {
55  // CD or CHDIR
56  return CMD_CD;
57  }
58  // this would be the COPY command
59  return CMD_SYNTAX;
60  break;
61  case 'M':
62  // MKDIR or MD
63  return CMD_MKDIR;
64  break;
65  case 'A':
66  return CMD_ASSIGN;
67  break;
68  case 'X':
69  // extensions for XD2031
70  return CMD_EXT;
71  break;
72  }
73  return CMD_SYNTAX;
74 }

115 // note: this does not return an actual error code,
116 // but only <0 if an error occurred; in that case, the error
117 // message must be set here.
118 int8_t command_execute(uint8_t channel_no, bus_t *bus, errormsg_t *errormsg,
119  void (*callback)(int8_t errnum, uint8_t *rxdata)) {
120 
121  cmd_t *command = &(bus->command);
122  rtconfig_t *rtconf = &(bus->rtconf);
123 
124  debug_printf("COMMAND: %s\n", (char*)&(command->command_buffer));
125 
126  parse_filename(command, &nameinfo, 1);
127 
128 #ifdef DEBUG_CMD
129  debug_printf("CMD=%s\n", nameinfo.cmd == CMD_NONE ? "-" : command_to_name(nameinfo.cmd));
130  debug_printf("DRIVE=%c\n", nameinfo.drive == 0xff ? '-' : nameinfo.drive + 0x30);
131  debug_printf("NAME=%s\n", (char*)nameinfo.name);
132  debug_puts("ACCESS="); debug_putc(isprint(nameinfo.access) ? nameinfo.access : '-'); debug_putcrlf();
133  debug_puts("TYPE="); debug_putc(isprint(nameinfo.type) ? nameinfo.type : '-'); debug_putcrlf();
134 #endif
135  // post-parse
136 
137  // pass-through commands
138  // those are just being passed to the provider
139  if (nameinfo.cmd == CMD_RENAME
140  || nameinfo.cmd == CMD_SCRATCH
141  || nameinfo.cmd == CMD_CD
142  || nameinfo.cmd == CMD_MKDIR
143  || nameinfo.cmd == CMD_RMDIR) {
144 
145  uint8_t type = 0;
146  switch(nameinfo.cmd) {
147  case CMD_RENAME:
148  type = FS_RENAME;
149  break;
150  case CMD_SCRATCH:
151  type = FS_DELETE;
152  break;
153  case CMD_CD:
154  type = FS_CHDIR;
155  break;
156  case CMD_MKDIR:
157  type = FS_MKDIR;
158  break;
159  case CMD_RMDIR:
160  type =FS_RMDIR;
161  break;
162  default:
163  // should not happen, all if() conditions are accounted for
164  debug_puts("ILLEGAL COMMAND!");
165  set_error(errormsg, ERROR_SYNTAX_UNKNOWN);
166  return ERROR_FAULT;
167  }
168 
169  return file_submit_call(channel_no, type, errormsg, rtconf, callback);
170  } else
171  if (nameinfo.cmd == CMD_ASSIGN) {
172 
173  if (nameinfo.drive == NAMEINFO_UNUSED_DRIVE) {
174  // no drive
175  set_error(errormsg, ERROR_DRIVE_NOT_READY);
176  return -1;
177  }
178 
179  // the +1 on the name skips the endpoint number stored in position 0
180  if (provider_assign(nameinfo.drive, (char*) nameinfo.name+1) < 0) {
181 
182  return file_submit_call(channel_no, FS_ASSIGN, errormsg, rtconf, callback);
183  } else {
184  // need to unlock the caller by calling the callback function
185  callback(ERROR_OK, NULL);
186  }
187  return 0;
188  } else
189  if (nameinfo.cmd == CMD_INITIALIZE) {
190  debug_puts("INITIALIZE");
191  // need to unlock the caller by calling the callback function
192  callback(ERROR_OK, NULL);
193  return 0;
194  } else
195  if (nameinfo.cmd == CMD_EXT) {
196  debug_puts("CONFIGURATION EXTENSION");
197  int8_t rv = rtconfig_set(rtconf, (char*) command->command_buffer);
198  callback(rv, NULL);
199  return 0;
200  }
201 
202  return -1;
203 }
