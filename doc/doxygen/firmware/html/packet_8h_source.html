42 /***********************************************************************************
43  * packet stuff
44  *
45  * A packet is a structure to communicate with the pc server. packets are sent and
46  * received via serial line.
47  *
48  * Logically and on the wire a packet contains a type, a packet length, and the data payload.
49  * The length on the wire contains the type and length bytes, so it is payload length plus two.
50  *
51  * A packet structure contains a buffer pointer, with length, read and write pointers, and
52  * a read/write state machine
53  *
54  * The internal length and read/write pointers work on the data payload, so packet.len is
55  * two bytes short of the length on the wire.
56  *
57  * Note that some types of packets contain the channel number as first byte of the payload.
58  * Also all FS_REPLY packets have this, so this knowledge is hardcoded in the uart code.
59  */
60

70 typedef struct {
71  uint8_t type; // one of the FS_* codes as packet type
72  uint8_t *buffer; // address of buffer
73  uint8_t wp; // write pointer (write to buffer[wp], then wp++)
74  uint8_t rp; // read pointer (if rp<wp then read from buffer[rp], then rp++)
75  uint8_t len; // length of buffer
76  int8_t chan; // channel number
77 } packet_t;
