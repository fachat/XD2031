78 static uint8_t *appendmsg(uint8_t *msg, const uint8_t *table, const uint8_t code) {
79 
80  uint8_t *ptr = (uint8_t*) table; // table index
81  uint8_t pattern = H(code);
82 
83  // first find the code in the give table
84  uint8_t c = rom_read_byte(ptr);
85  while (c != 0 && c != pattern) {
86  ptr++;
87  c = rom_read_byte(ptr);
88  }
89  if (c == 0) {
90  // not found
91  return msg;
92  }
93  // skip over remaining codes
94  c = rom_read_byte(ptr);
95  while (c & 0x80) {
96  ptr++;
97  c = rom_read_byte(ptr);
98  }
99  // now ptr points to the first text entry
100  // c contains the value
101  do {
102  if (c < 32) {
103  // get text from token table (recursive)
104  msg = appendmsg(msg, tokens, c);
105  } else {
106  *msg = c;
107  msg++;
108  }
109  ptr++;
110  c = rom_read_byte(ptr);
111  } while (c != 0 && (c & 0x80) == 0);
112 
113  return msg;
114 }

116 /* Append a decimal number to a string */
117 uint8_t *appendnumber(uint8_t *msg, uint8_t value) {
118  if (value >= 100) {
119  *msg++ = '0' + value/100;
120  value %= 100;
121  }
122  *msg++ = '0' + value/10;
123  *msg++ = '0' + value%10;
124 
125  return msg;
126 }

128 void set_error_ts(errormsg_t *err, uint8_t errornum, uint8_t track, uint8_t sector) {
129  uint8_t *msg = err->error_buffer;
130 
131  err->readp = 0;
132 
133  msg = appendnumber(msg,errornum);
134  *msg++ = ',';
135 
136  if (errornum == ERROR_DOSVERSION) {
137  char *p = (char*) versionstr;
138  uint8_t c;
139  while ((c = rom_read_byte(p++)) != 0) {
140  *msg++ = c;
141  }
142  } else {
143  msg = appendmsg(msg,messages,errornum);
144  }
145  *msg++ = ',';
146 
147  msg = appendnumber(msg,track);
148  *msg++ = ',';
149 
150  msg = appendnumber(msg,sector);
151  *msg++ = 13;
152 
153  // end string marker
154  *msg = 0;
155 
156  if (errornum != ERROR_OK && errornum != ERROR_DOSVERSION && errornum != ERROR_SCRATCHED) {
157  led_set(ERROR);
158  } else {
159  // same as idle, but clears error
160  led_set(OFF);
161  }
162 
163 #ifdef DEBUG_ERROR
164 debug_printf("Set status to: %s\n", err->error_buffer);
165 #endif
166 }
