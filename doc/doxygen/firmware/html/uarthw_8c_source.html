75 int8_t uarthw_can_send() {
76  // check for wrap-around to read pointer
77  uint8_t tmp = BUFFER_NEXT(tx_wp);
78  return (tmp != tx_rp) ? 1 : 0;

84 void uarthw_send(int8_t data) {
85  txbuf[tx_wp] = data;
86  tx_wp = BUFFER_NEXT(tx_wp);
87  _uart_enable_sendisr();

94 int16_t uarthw_receive() {
95  if (rx_wp == rx_rp) {
96  // no data available
97  return -1;
98  }
99  int16_t data = 0xff & rxbuf[rx_rp];
100  rx_rp = BUFFER_NEXT(rx_rp);
101  return data;
102 }
