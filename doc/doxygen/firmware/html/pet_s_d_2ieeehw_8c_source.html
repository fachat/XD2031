43 static void ieee_interrupts_init(void) {
44  DDRD &= ~_BV(PD2); // define ATN as input
45  PORTD |= _BV(PD2); // enable pull-up
46 
47  EIMSK &= (uint8_t) ~_BV(INT0); // disable interrupt
48  EICRA &= (uint8_t) ~_BV(ISC00); // configure interrupt on falling
49  EICRA |= _BV(ISC01); // edge of ATN
50  EIMSK |= _BV(INT0); // enable interrupt
51  //debug_putps("Done init ieee ints"); debug_putcrlf();
52 }
53 
54 /* IEEE-488 ATN interrupt using INT0 */
55 static void set_atn_irq(uint8_t x) {
56 #if DEBUG
57  debug_putps("ATN_IRQ:"); debug_puthex(x); debug_putcrlf();
58 #endif
59  if (x)
60  EIMSK |= _BV(IEEE_ATN_INT);
61  else
62  EIMSK &= (uint8_t) ~_BV(IEEE_ATN_INT);
63 }

79 void ieeehw_init() {
80 
81  IEEE_DDR_TE |= _BV(IEEE_PIN_TE); // Define TE as output
82  IEEE_PORT_TE &= (uint8_t) ~ _BV(IEEE_PIN_TE); // TE=0
83  IEEE_DDR_DC |= _BV(IEEE_PIN_DC); // Define DC as output
84  IEEE_PORT_DC |= _BV(IEEE_PIN_DC); // DC=1 (device)
85 
86  // disable ATN interrupt
87  set_atn_irq(0);

91 
92  // start ATN interrupt handling
93  ieee_interrupts_init();
