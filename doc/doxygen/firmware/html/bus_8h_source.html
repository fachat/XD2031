28 /*
29  * IEEE488 impedance layer
30  *
31  * provides the parallelsendbyte, parallelreceivebyte and parallelattention
32  * methods called by the IEEE loop, and translates these into
33  * calls to the channel framework, open calls etc.
34  */
35 
36 // these are the runtime variables for a bus end point like
37 // the IEEE488 or the serial IEC bus.
38 typedef struct {
39  // configuration
40  uint8_t current_device_address; // current device address (copied from rtconfig)
41  uint8_t secaddr_offset; // offset to use on secondary address to get channel no
42  // to avoid collisions with other busses
43  // runtime for the bus
44  channel_t *channel; // current open file channel
45 
46  uint8_t device; // primary command byte, includes dev addr and TALK/LISTEN/...
47  uint8_t secondary; // secondary command byte, includes sec addr and OPEN/CLOSE/...
48 
49  // interrupt callback handling for commands and opens
50  uint8_t cmd_done; // set on a callback from the irq
51  int8_t errnum; // from interrupt between BUS_CMDWAIT and BUS_CMDPROCESS
52  uint8_t errparam; // from interrupt between BUS_CMDWAIT and BUS_CMDPROCESS
53 
54  // command channel
55  cmd_t command; // command buffer
56  //errormsg_t error; // error message - is currently shared between busses
57 
58  // runtime config, like unit number, last used drive etc
59  rtconfig_t rtconf;
60 } bus_t;
