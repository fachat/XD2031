48 static void send() {
49 
50  // tell the packet what's in the buffer to be read
51  // nchars are in the buffer to be sent; channel is -1 (for terminal)
52  packet_set_filled(&termpack, 0xfe, FS_TERM, nchars);
53  endpoint->provider->submit(endpoint->provdata, &termpack);
54  nchars = 0;
55 }

57 void term_flush() {
58  if (nchars > 0) {
59  // has not been sent with the last char
60  // flush
61  if (endpoint != NULL) {
62  send();
63  }
64  }
65  serial_flush();
66 }

68 void term_putc(char c) {
69 
70  // wait if buffer is still being sent
71  packet_wait_done(&termpack);
72 
73  if (nchars < TERM_BUFFER_LENGTH) {
74  buf[nchars] = c;
75  nchars++;
76  }
77 
78  if (nchars >= TERM_BUFFER_LENGTH) {
79  if (endpoint != NULL) {
80  send();
81  }
82  }
83 }

101 void term_rom_puts(const char *rom_str) {
102 
103 
104  packet_wait_done(&termpack);
105 
106  uint8_t len = rom_strlen(rom_str);
107  if (len >= TERM_BUFFER_LENGTH) {
108  // cut, can't send more bytes anyway
109  len = TERM_BUFFER_LENGTH;
110  }
111 
112  if ((nchars + len) >= TERM_BUFFER_LENGTH) {
113  if (endpoint == NULL) {
114  // no way we can print that now
115  return;
116  }
117  send();
118  }
119  packet_wait_done(&termpack);
120  rom_memcpy(buf+nchars, rom_str, len);
121 
122  nchars += len;
123 }
