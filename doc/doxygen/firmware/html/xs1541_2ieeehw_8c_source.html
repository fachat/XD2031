38 static void ieee_interrupts_init(void) {
39  /* clear interrupt flag */
40  PCIFR |= _BV(PCIF3);
41 
42  /* enable ATN in pin change enable mask
43  * translates to
44  * PCMSK3 |= _BV(PCINT27)
45  * which is ok for PD3
46  */
47  IEEE_PCMSK |= _BV(IEEE_PCINT);
48 
49  /* Enable pin change interrupt 3 (PCINT31..24) */
50  PCICR |= _BV(PCIE3);
51 
52  //debug_putps("Done init ieee ints"); debug_putcrlf();
53 }

56 static void set_atn_irq(uint8_t x) {
57 #if DEBUG
58  debug_putps("ATN_IRQ:"); debug_puthex(x); debug_putcrlf();
59 #endif
60  if (x)

63  IEEE_PCMSK &= (uint8_t) ~_BV(IEEE_PCINT);
