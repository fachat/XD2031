80 static inline void ndachi() {
81  // disable interrupt to avoid race condition
82  // of ATN irq between the atnishi() check and
83  // setting NDAC lo
84  cli();
85  if (atnishi() || is_atna) {
86  IEEE_DDR_NDAC &= (uint8_t)~_BV(IEEE_PIN_NDAC); // NDAC as input
87  IEEE_PORT_NDAC |= _BV(IEEE_PIN_NDAC); // Enable pull-up
88  }
89  // allow interrupt again
90  sei();
91  is_ndacout = 1;
92 }
