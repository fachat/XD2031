61 // opens the file, registers an error code in command->error if necessary
62 // If the open was successful, setup a channel for the given channel number
63 // (note: explicitely not secondary device number, as IEEE and IEC in parallel
64 // use overlapping numbers, so they may be shifted or similar to avoid clashes)
65 //
66 // The command buffer is used as transmit buffer, so it must not be overwritten
67 // until the open has been sent.
68 //
69 int8_t file_open(uint8_t channel_no, bus_t *bus, errormsg_t *errormsg,
70  void (*callback)(int8_t errnum, uint8_t *rxdata), uint8_t is_save) {
71 
72  assert_not_null(bus, "file_open: bus is null");
73 
74  cmd_t *command = &(bus->command);
75  rtconfig_t *rtconf = &(bus->rtconf);
76 
77 #ifdef DEBUG_FILE
78  debug_printf("OPEN FILE: FOR CHAN: %d WITH NAME: %s\n",
79  channel_no, (char*)&(command->command_buffer));
80 #endif
81 
82  // note: in a preemtive env, the following would have to be protected
83  // to be atomic as we modify static variables
84 
85  parse_filename(command, &nameinfo, 0);
86 
87  // post-parse
88 
89  if (nameinfo.cmd != CMD_NONE && nameinfo.cmd != CMD_DIR) {
90  // command name during open
91  // this is in fact ignored by CBM DOS as checked with VICE's true drive emulation
92  debug_printf("NO CORRECT CMD: %s\n", command_to_name(nameinfo.cmd));
93  nameinfo.cmd = 0;
94  }
95  if (nameinfo.type != 0 && nameinfo.type != 'S' && nameinfo.type != 'P') {
96  // not set, or set as not sequential and not program
97  debug_puts("UNKOWN FILE TYPE: "); debug_putc(nameinfo.type); debug_putcrlf();
98  set_error(errormsg, ERROR_FILE_TYPE_MISMATCH);
99  return -1;
100  }
101  if (nameinfo.access != 0 && nameinfo.access != 'W' && nameinfo.access != 'R'
102  && nameinfo.access != 'A') {
103  debug_puts("UNKOWN FILE ACCESS TYPE "); debug_putc(nameinfo.access); debug_putcrlf();
104  // not set, or set as not read, write, or append
105  set_error(errormsg, ERROR_SYNTAX_UNKNOWN);
106  return -1;
107  }
108  if (nameinfo.cmd == CMD_DIR && (nameinfo.access != 0 && nameinfo.access != 'R')) {
109  // trying to write to a directory
110  debug_puts("WRITE TO DIRECTORY!"); debug_putcrlf();
111  set_error(errormsg, ERROR_FILE_EXISTS);
112  return -1;
113  }
114 
115  uint8_t type = FS_OPEN_RD;
116  // either ",W" or secondary address is one, i.e. save
117  if (nameinfo.access == 'W' || is_save) type = FS_OPEN_WR;
118  if (nameinfo.access == 'A') type = FS_OPEN_AP;
119  if (nameinfo.cmd == CMD_DIR) type = FS_OPEN_DR;
120 
121 
122  return file_submit_call(channel_no, type, errormsg, rtconf, callback);
123 
124 }

126 uint8_t file_submit_call(uint8_t channel_no, uint8_t type, errormsg_t *errormsg, rtconfig_t *rtconf,
127  void (*callback)(int8_t errnum, uint8_t *rxdata)) {
128 
129  assert_not_null(errormsg, "file_submit_call: errormsg is null");
130  assert_not_null(rtconf, "file_submit_call: rtconf is null");
131 
132  // check for default drive (here is the place to set the last used one)
133  if (nameinfo.drive == 0xff) {
134  // there currently only is a single drive, 0
135  nameinfo.drive = rtconf->last_used_drive;
136  // TODO: fix this hack!
137  nameinfo.name[0] = rtconf->last_used_drive;
138  }
139 
140  // here is the place to plug in other file system providers,
141  // like SD-Card, or even an outgoing IEC or IEEE, to convert between
142  // the two bus systems. This is done depending on the drive number
143  // and managed with the ASSIGN call.
144  //provider_t *provider = &serial_provider;
145  endpoint_t *endpoint = provider_lookup(nameinfo.drive);
146 
147  // check the validity of the drive (note that in general provider_lookup
148  // returns a default provider - serial-over-USB to the PC, which then
149  // may do further checks
150  if (endpoint == NULL) {
151  debug_puts("ILLEGAL DRIVE: "); debug_putc(0x30+nameinfo.drive); debug_putcrlf();
152  set_error(errormsg, ERROR_DRIVE_NOT_READY);
153  return -1;
154  }
155  provider_t *provider = endpoint->provider;
156 
157  // find open slot
158  //int8_t slot = -1;
159  open_t *activeslot = NULL;
160  for (uint8_t i = 0; i < MAX_ACTIVE_OPEN; i++) {
161  if (active[i].channel_no < 0) {
162  //slot = i;
163  activeslot = (open_t*) &active[i];
164  break;
165  }
166  }
167  //if (slot < 0) {
168  if (activeslot == NULL) {
169  debug_puts("NO OPEN SLOT FOR OPEN!");
170  debug_putcrlf();
171  set_error(errormsg, ERROR_NO_CHANNEL);
172  return -1;
173  }
174 
175  packet_init(&activeslot->txbuf, nameinfo.namelen, nameinfo.name);
176  packet_set_filled(&activeslot->txbuf, channel_no, type, nameinfo.namelen);
177 
178  if (provider->to_provider != NULL && provider->to_provider(&activeslot->txbuf) < 0) {
179  // converting the file name to the provider exceeded the buffer space
180  debug_puts("NAME CONVERSION EXCEEDS BUFFER!");
181  debug_putcrlf();
182  set_error(errormsg, ERROR_SYNTAX_NONAME);
183  return -1;
184  }
185 
186  // open channel
187  uint8_t writetype = (type == FS_OPEN_WR || type == FS_OPEN_AP) ? 1 : 0;
188  int8_t (*converter)(packet_t*) = (type == FS_OPEN_DR) ? (provider->directory_converter) : NULL;
189 
190  channel_t *channel = channel_find(channel_no);
191  if (channel != NULL) {
192  debug_puts("FILE OPEN ERROR");
193  debug_putcrlf();
194  set_error(errormsg, ERROR_NO_CHANNEL);
195  // clean up
196  channel_close(channel_no);
197  return -1;
198  }
199 
200  int8_t e = channel_open(channel_no, writetype, endpoint, converter);
201  if (e < 0) {
202  debug_puts("E="); debug_puthex(e); debug_putcrlf();
203  set_error(errormsg, ERROR_NO_CHANNEL);
204  return -1;
205  }
206 
207  activeslot->callback = callback;
208 
209  // no more error here, just the submit.
210  // so callers can be sure if this function returns <0, they do not need
211  // to close the channel, as it has not been opened
212  // If this function returns 0, a callback must be received and handled,
213  // and the channel is already opened.
214 
215  activeslot->channel_no = channel_no;
216 
217  // prepare response buffer
218  packet_init(&activeslot->rxbuf, OPEN_RX_DATA_LEN, activeslot->rxdata);
219 
220  provider->submit_call(endpoint->provdata, channel_no, &activeslot->txbuf,
221  &activeslot->rxbuf, _file_open_callback);
222 
223  return 0;
224 }

226 static void _file_open_callback(int8_t channelno, int8_t errnum) {
227 
228  // callback to opener
229  // free data structure for next open
230  for (uint8_t i = 0; i < MAX_ACTIVE_OPEN; i++) {
231  if (active[i].channel_no == channelno) {
232  if (errnum < 0) {
233  // we did not receive the packet!
234  active[i].callback(ERROR_DRIVE_NOT_READY, NULL);
235  } else {
236  // we did receive the reply packet
237  // NOTE: rxdata[0] is actually rxdata[FSP_DATA], as first
238  // byte of reply packet is the error number
239  // Note that the reply packet already sends an official errors.h
240  // error code, so no translation needed
241  active[i].callback(active[i].rxdata[0], active[i].rxdata);
242  }
243  active[i].channel_no = -1;
244  break;
245  }
246  }
247 }
