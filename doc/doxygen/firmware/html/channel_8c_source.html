98  * open and reserve a channel for the given chan channel number
99  * returns negative on error, 0 on ok
100  *
101  * writetype is either 0 for read only, 1 for write, (as seen from ieee device)

103 int8_t channel_open(int8_t chan, uint8_t writetype, endpoint_t *prov, int8_t (*dirconv)(packet_t *)) {
104 
105 //debug_printf("channel_open: chan=%d\n", chan);
106 
107  for (int8_t i = MAX_CHANNELS-1; i>= 0; i--) {
108  if (channels[i].channel_no < 0) {
109  channels[i].channel_no = chan;
110  channels[i].current = 0;
111  channels[i].writetype = writetype;
112  channels[i].endpoint = prov;
113  channels[i].directory_converter = dirconv;
114  channels[i].pull_state = PULL_OPEN;
115  channels[i].push_state = PUSH_OPEN;
116  // note: we should not channel_pull() here, as file open has not yet even been sent
117  // the pull is done in the open callback for a read-only channel
118  for (uint8_t j = 0; j < 2; j++) {
119  packet_reset(&channels[i].buf[j], chan);
120  }
121  return 0;
122  }
123  }
124  return -1;
125 }

136 static void channel_preload_int(channel_t *chan, uint8_t wait) {
137 
138  // TODO:fix for read/write
139  if (chan->writetype != WTYPE_READONLY) return;
140 
141  do {
142  if (chan->pull_state == PULL_OPEN) {
143  chan->current = 0;
144  channel_pull(chan, 0);
145  }
146  if (wait) {
147  while (chan->pull_state == PULL_PRELOAD) {
148  delayms(1);
149  }
150  }
151  if (chan->pull_state == PULL_ONECONV) {
152  packet_t *curpack = &(chan->buf[chan->current]);
153  if ((!packet_has_data(curpack)) && (!packet_is_last(curpack))) {
154  // zero length packet received
155  chan->pull_state = PULL_OPEN;
156  } else {
157  if (chan->directory_converter != NULL) {
158  //debug_printf(">>1: %p, p=%p\n",chan->directory_converter, &chan->buf[chan->current]);
159  //debug_printf(">>1: b=%p\n", packet_get_buffer(&chan->buf[chan->current]));
160  chan->directory_converter(&chan->buf[chan->current]);
161  }
162  chan->pull_state = PULL_ONEREAD;
163  }
164  }
165  if (chan->pull_state == PULL_TWOCONV) {
166  packet_t *opack = &(chan->buf[1-chan->current]);
167  if ((!packet_has_data(opack)) && (!packet_is_last(opack))) {
168  // zero length packet received
169  chan->pull_state = PULL_ONEREAD;
170  } else {
171  if (chan->directory_converter != NULL) {
172  //debug_printf(">>2: %p, p=%p\n",chan->directory_converter, &chan->buf[1-chan->current]);
173  //debug_printf(">>2: b=%p\n", packet_get_buffer(&chan->buf[1-chan->current]));
174  chan->directory_converter(&chan->buf[1-chan->current]);
175  }
176  chan->pull_state = PULL_TWOREAD;
177  }
178  }
179  }
180  while (chan->pull_state == PULL_OPEN);
181 //led_on();
182 }

250 void channel_close(int8_t channel_no) {
251  channel_t *chan = channel_find(channel_no);
252 
253  debug_printf("channel_close(%p -> %d), push_state=%d\n", chan, channel_no, chan->push_state); debug_flush();
254 
255  if (chan != NULL) {
256  if (chan->push_state != PUSH_OPEN) {
257  // if it's not PUSH_FILLONE, then it is in the process
258  // of being pushed
259  while (chan->push_state != PUSH_FILLONE) {
260  delayms(1);
261  }
262 
263  packet_t *curpack = &chan->buf[chan->current];
264  int l = packet_get_contentlen(curpack);
265 
266  // even if l==0, send an EOF packet to close the file
267  packet_set_filled(curpack, channel_no, FS_EOF, l);
268 
269  endpoint_t *endpoint = chan->endpoint;
270  endpoint->provider->submit_call(endpoint->provdata,
271  channel_no, curpack, curpack, _push_callback);
272 
273  // wait until the packet has been sent and been responded to
274  while (chan->push_state == PUSH_FILLONE) {
275  delayms(1);
276  }
277  }
278  channel_close_int(chan, 0);
279  }

287 channel_t* channel_refill(channel_t *chan) {
288  // buf->refill();
289  // buf = find_buffer(...)
290  //
291  uint8_t other = 1-chan->current;
292  if (!packet_is_last(&chan->buf[chan->current])) {
293  // current packet is not last one
294  // other packet should have been pulled in channel_next()
295  // so it is either empty (request, or FS_EOF), or has data
296  packet_t *opacket = &chan->buf[other];
297 
298  // wait until available
299  while(chan->pull_state == PULL_PULL2ND) {
300  delayms(1);
301  }
302  if (chan->pull_state == PULL_TWOCONV) {
303  if (chan->directory_converter != NULL) {
304 //debug_printf(">>3: %p, p=%p\n",chan->directory_converter, &chan->buf[1-chan->current]);
305 //debug_printf(">>3: b=%p\n", packet_get_buffer(&chan->buf[1-chan->current]));
306  chan->directory_converter(&chan->buf[1-chan->current]);
307  }
308  chan->pull_state = PULL_TWOREAD;
309  }
310 
311  if (packet_has_data(opacket)) {
312  // switch packets
313  chan->current = other;
314 
315  chan->pull_state = PULL_ONEREAD;
316 
317  if (!packet_is_last(&chan->buf[chan->current])) {
318  channel_pull(chan, 1-other);
319  }
320  return chan;
321  }
322  }
323 
324  // no more data - close it
325  channel_close_int(chan, 0);
326  return NULL;
327 }

329 static void _push_callback(int8_t channelno, int8_t errnum) {
330  channel_t *p = channel_find(channelno);
331  if (p != NULL) {
332  p->last_push_errorno = errnum;
333 
334  // TODO: only if errorno == 0?
335  // Probably need some PUSH_ERROR as well
336  if (p->push_state == PUSH_FILLTWO) {
337  // release possibly waiting channel_put()
338  p->push_state = PUSH_FILLONE;
339  } else
340  if (p->push_state == PUSH_FILLONE) {
341  // release possibly waiting channel_close()
342  p->push_state = PUSH_OPEN;
343  }
344  }
345 }

348 channel_t* channel_put(channel_t *chan, char c, int forceflush) {
349 
350  if (chan->push_state == PUSH_OPEN) {
351  chan->push_state = PUSH_FILLONE;
352  }
353 
354  uint8_t channo = chan->channel_no;
355  packet_t *curpack = &chan->buf[chan->current];
356 
357  packet_write_char(curpack, (uint8_t) c);
358 
359  if (packet_is_full(curpack) || forceflush) {
360  packet_set_filled(curpack, channo, FS_WRITE, packet_get_contentlen(curpack));
361 
362  // wait until the other packet has been replied to,
363  // i.e. it has been sent, the buffer is free again
364  // which we need for the next channel_put
365  while (chan->push_state == PUSH_FILLTWO) {
366  delayms(1);
367  //serial_delay();
368  }
369 
370  // note that we pushed one and are now filling the second
371  chan->push_state = PUSH_FILLTWO;
372 
373  // use same packet as rx/tx buffer
374  endpoint_t *endpoint = chan->endpoint;
375  endpoint->provider->submit_call(endpoint->provdata,
376  channo, curpack, curpack, _push_callback);
377 
378  // switch
379  chan->current = 1-chan->current;
380  packet_reset(&chan->buf[chan->current], channo);
381  }
382  return chan;
383 }
